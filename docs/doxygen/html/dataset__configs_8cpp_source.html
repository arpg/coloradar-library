29std::vector<std::string> DatasetExportConfig::parseRuns(const YAML::Node &runsValue) {
30 if (!runsValue || runsValue.IsNull()) return {};
31 std::vector<std::string> runs;
32
33 if (runsValue.IsScalar()) {
34 std::string run = runsValue.as<std::string>();
35 runs.push_back(run);
36 } else if (runsValue.IsSequence()) {
37 for (const auto &n : runsValue) {

39 std::string run = n.as<std::string>();

43 return validateRuns(runs);

47std::filesystem::path DatasetExportConfig::validateDestination(const std::filesystem::path &destination) {
48 std::filesystem::path absoluteDestination = std::filesystem::absolute(destination);
49 std::filesystem::path directoriesPath = absoluteDestination.parent_path();
50 std::string filename = absoluteDestination.filename().string();
51 if (!std::filesystem::exists(directoriesPath)) {
52 std::filesystem::create_directories(directoriesPath);
53 }
54 if (filename.empty()) {
55 auto now = std::chrono::system_clock::now();
56 std::time_t now_c = std::chrono::system_clock::to_time_t(now);
57 char buffer[100];
58 std::strftime(buffer, sizeof(buffer), "%Y%m%d_%H%M%S", std::localtime(&now_c));
59 filename = "dataset_" + std::string(buffer) + ".h5";
60 }
61 if (std::filesystem::path(filename).extension() != ".h5") {
62 filename += ".h5";
63 }
64 return directoriesPath / filename;

67std::vector<std::string> DatasetExportConfig::validateRuns(const std::vector<std::string> &runs) {
68 if (runs.size() == 1 && (runs[0].empty() || runs[0] == "all")) return {};
69 for (const auto &run : runs) {
70 if (run.empty()) throw std::runtime_error("Invalid run name: empty string.");
71 if (run == "all") throw std::runtime_error("Invalid run name: 'all' is not allowed if other names are specified.");
72 for (unsigned char c : run) {
73 if (!(std::isalnum(c) || c == '_' || c == '-')) throw std::runtime_error(
74 "Invalid run name \"" + run + "\"\nOnly letter, number, '_' or '-' characters are allowed."
75 "\nUse correct YAML list formats:\n```\n runs:\n - run1\n - run2\n```\nor as a flow sequence:\n```\n runs: [run1, run2]\n```"
76 );
77 }
78 }
79 return runs;
