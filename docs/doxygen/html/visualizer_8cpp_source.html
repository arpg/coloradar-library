62void DatasetVisualizer::visualize(const std::shared_ptr<Run> run, const bool usePrebuiltMap) {
63 reset();
64 this->run = run;
65
66 // init poses and use cascade timestamps as reference
67 auto basePoses = run->getPoses<Eigen::Affine3f>();
68 if (basePoses.empty()) throw std::runtime_error("Base poses are empty.");
69 basePosesLidarTs = interpolatePoses(basePoses, run->poseTimestamps(), run->lidarTimestamps());
70 basePosesCascadeTs = interpolatePoses(basePoses, run->poseTimestamps(), run->cascadeTimestamps());
71 cascadeToLidarFrameIndices.resize(basePosesCascadeTs.size());
72 for (size_t cascadeIdx = 0; cascadeIdx < basePosesCascadeTs.size(); cascadeIdx++) {
73 auto cascadeTimestamp = run->cascadeTimestamps()[cascadeIdx];
74 auto lidarIdx = findClosestTimestampIndex(cascadeTimestamp, run->lidarTimestamps());
75 cascadeToLidarFrameIndices[cascadeIdx] = lidarIdx;
76 }
77 numSteps = basePosesCascadeTs.size();
78 mapIsPrebuilt = usePrebuiltMap;
79 if (usePrebuiltMap) {
80 lidarMapCloud = run->getLidarOctomap();
81 if (lidarMapCloud->empty()) throw std::runtime_error("Prebuilt map cloud is empty.");
82 filterOccupancy(lidarMapCloud, 0, true);
83 renderLidarMap();
84 }
85
86 // display frame 0
87 step(1);
88
89 // start viewer
90 viewer->setBackgroundColor(0, 0, 0);
91 viewer->addCoordinateSystem(2.0);
92 if (std::filesystem::exists(cameraConfigPath)) {
93 viewer->loadCameraParameters(cameraConfigPath);
94 showMessage("Camera parameters loaded from " + std::filesystem::absolute(cameraConfigPath).string());
95 }
96 viewer->registerKeyboardCallback(std::bind(&DatasetVisualizer::keyboardCallback, this, std::placeholders::_1));
97 renderLegend();
98 while (!viewer->wasStopped()) {
99 viewer->spin();
100 }
101}

166void DatasetVisualizer::step(const int increment) {
167 if (increment == 0) throw std::runtime_error("Step increment must be non-zero.");
168 int prevStep = currentStep;
169 int targetStep = std::clamp(currentStep + increment, 0, numSteps - 1);
170 if (targetStep == currentStep) return;
171 currentStep = targetStep;
172
173 if (!mapIsPrebuilt) {
174 // update lidar map if accumulating
175 updateLidarMap(currentStep);
176 renderLidarMap();
177 }
178
179 auto cascadeCloud = readCascadeCloud(currentStep);
180 renderRadarCloud(cascadeCloud);
181 renderLegend();
182}

185void DatasetVisualizer::updateLidarMap(const int step) {
186 if (step >= numSteps) throw std::runtime_error("Step " + std::to_string(step) + " is out of range [0, " + std::to_string(numSteps) + ").");
187 if (currentMapStep >= step) return;
188 for (int i = currentMapStep + 1; i <= step; i++) {
189 int lidarFrameIdx = cascadeToLidarFrameIndices[i];
190 auto lidarCloud = readLidarCloud(lidarFrameIdx);
191 lidarCloud = downsampleLidarCloud(lidarCloud);
192 *lidarMapCloud += *lidarCloud;
193 }
194 currentMapStep = step;
195}

198pcl::PointCloud<RadarPoint>::Ptr DatasetVisualizer::readCascadeCloud(const int scanIdx) {
199 auto cascadeHeatmap = run->getCascadeHeatmap(scanIdx);
200 int targetNumRangeBins = static_cast<int>(cascadeRadarConfig->nRangeBins() * 0.75);
201 if (cascadeRadarConfig->numElevationBins() > 0 || cascadeRadarConfig->nRangeBins() > targetNumRangeBins) {
202 auto result = cascadeRadarConfig->clipHeatmap(cascadeHeatmap, cascadeRadarConfig->numAzimuthBins(), 0, targetNumRangeBins);
203 cascadeHeatmap = result.heatmap;
204 }
205 auto cascadeCloud = clippedCascadeRadarConfig->heatmapToPointcloud(cascadeHeatmap, cascadeRadarIntensityThreshold); // use clipped config to convert heatmap to pointcloud
206 cascadeCloud = normalizeRadarCloudIntensity(cascadeCloud);
207
208 auto map_T_base = basePosesCascadeTs[scanIdx]; // read map_T_base as "base to map"
209 auto base_T_cascade = baseToCascadeTransform; // read base_T_cascade as "cascade to base"
210 auto map_T_cascade = map_T_base * base_T_cascade; // read map_T_cascade as "cascade to map"
211 pcl::transformPointCloud(*cascadeCloud, *cascadeCloud, map_T_cascade); // map frame
212 return cascadeCloud;
213}

236pcl::PointCloud<RadarPoint>::Ptr DatasetVisualizer::normalizeRadarCloudIntensity(const pcl::PointCloud<RadarPoint>::Ptr& inputCloud) const {
237 auto normalizedCloud = std::make_shared<pcl::PointCloud<RadarPoint>>();
238 normalizedCloud->reserve(inputCloud->size());
239
240 float minIntensity = std::numeric_limits<float>::max();
241 float maxIntensity = std::numeric_limits<float>::lowest();
242 for (const auto& point : *inputCloud) {
243 if (point.intensity < minIntensity) minIntensity = point.intensity;
244 if (point.intensity > maxIntensity) maxIntensity = point.intensity;
245 }
246 float range = maxIntensity - minIntensity;
247 if (range == 0) range = 1.0f;
248
249 for (const auto& point : *inputCloud) {
250 RadarPoint normalizedPoint = point;
251 normalizedPoint.intensity = (point.intensity - minIntensity) / range;
252 normalizedCloud->push_back(normalizedPoint);
253 }
254 return normalizedCloud;
255}

258pcl::PointCloud<RadarPoint>::Ptr DatasetVisualizer::extractTopNIntensity(const pcl::PointCloud<RadarPoint>::Ptr& inputCloud, size_t N) const {
259 auto sorted = *inputCloud;
260 std::nth_element(sorted.begin(), sorted.begin() + N, sorted.end(),
261 [](const RadarPoint& a, const RadarPoint& b) { return a.intensity > b.intensity; });
262 sorted.resize(std::min(N, sorted.size()));
263
264 auto output = std::make_shared<pcl::PointCloud<RadarPoint>>();
265 output->insert(output->end(), sorted.begin(), sorted.end());
266 return output;
267}
