29struct PointKey {

34 PointKey(float x, float y, int precision)

51template <Pcl4dPointType PointT, template <typename> class CloudT>
52void collapseElevation(std::shared_ptr<CloudT<PointT>>& cloud, const float& elevationMinMeters, const float& elevationMaxMeters) {
53 const int precision = 4;
54 if (elevationMinMeters > elevationMaxMeters)
55 throw std::invalid_argument("Invalid elevation range: elevationMin must be less or equal to elevationMax.");
56 std::unordered_map<PointKey, PointT, PointKeyHash> maxIntensityMap;
57 maxIntensityMap.reserve(cloud->size());
58 for (const auto& point : *cloud) {
59 if (point.z >= elevationMinMeters && point.z <= elevationMaxMeters) {
60 PointKey key(point.x, point.y, precision);
61 auto iter = maxIntensityMap.find(key);
62 if (iter == maxIntensityMap.end() || point.intensity > iter->second.intensity) {
63 maxIntensityMap[key] = point;
64 }
65 }
66 }
67 cloud->clear();
68 cloud->reserve(maxIntensityMap.size());
69 for (const auto& [key, point] : maxIntensityMap) {
70 PointT collapsedPoint = point;
71 collapsedPoint.z = 0.0f;
72 cloud->push_back(collapsedPoint);
73 }
74}

98template <Pcl4dPointType PointT, template <typename> class CloudT>
99void filterOccupancy(std::shared_ptr<CloudT<PointT>>& cloud, const float& probabilityThreshold, const bool& saveProbabilities) {
100 if (probabilityThreshold < 0.0f || probabilityThreshold > 1.0f) {
101 throw std::out_of_range("Invalid probability threshold: expected value from 0 to 1, got " + std::to_string(probabilityThreshold));
102 }
103 if (probabilityThreshold == 0.0f && !saveProbabilities) return;
104
105 CloudT<PointT> filteredCloud;
106 filteredCloud.reserve(cloud->size());
107 for (const auto& point : *cloud) {
108 float probability = 1.0f / (1.0f + std::exp(-point.intensity));
109 if (probability >= probabilityThreshold) {
110 PointT newPoint = point;
111 if (saveProbabilities) newPoint.intensity = probability;
112 filteredCloud.push_back(newPoint);
113 }
114 }
115 cloud->swap(filteredCloud);
116}

119template<PoseType PoseT>
120std::vector<PoseT> interpolatePoses(const std::vector<PoseT>& poses, const std::vector<double>& poseTimestamps, const std::vector<double>& targetTimestamps) {
121 if (poses.empty()) throw std::runtime_error("Cannot interpolate empty poses.");
122 if (poseTimestamps.empty()) throw std::runtime_error("Cannot interpolate over empty pose timestamps.");
123 if (targetTimestamps.empty()) throw std::runtime_error("Cannot interpolate over empty target timestamps.");
124 if (poses.size() != poseTimestamps.size()) throw std::runtime_error("The length of poses and poses' timestamps vectors must match.");
125
126 std::vector<PoseT> interpolatedPoses;
127 size_t tsStartIdx = 0, tsEndIdx = poseTimestamps.size() - 1;
128 size_t tsIdx = tsStartIdx;
129
130 for (size_t targetTsIdx = 0; targetTsIdx < targetTimestamps.size(); ++targetTsIdx) {
131 if (targetTimestamps[targetTsIdx] < poseTimestamps[tsStartIdx]) {
132 interpolatedPoses.push_back(poses[tsStartIdx]);
133 continue;
134 }
135 if (targetTimestamps[targetTsIdx] > poseTimestamps[tsEndIdx]) {
136 interpolatedPoses.push_back(poses[tsEndIdx]);
137 continue;
138 }
139 while (tsIdx + 1 <= tsEndIdx && poseTimestamps[tsIdx + 1] < targetTimestamps[targetTsIdx])
140 tsIdx++;
141 double denominator = poseTimestamps[tsIdx + 1] - poseTimestamps[tsIdx];
142 double ratio = denominator > 0.0f ? (targetTimestamps[targetTsIdx] - poseTimestamps[tsIdx]) / denominator : 0.0f;
143
144 Eigen::Vector3f t1 = coloradar::internal::toEigenTrans(poses[tsIdx]);
145 Eigen::Vector3f t2 = coloradar::internal::toEigenTrans(poses[tsIdx + 1]);
146 Eigen::Vector3f interpolatedTransEig = (1.0f - ratio) * t1 + ratio * t2;
147 auto interpolatedTrans = coloradar::internal::fromEigenTrans<typename PoseTraits<PoseT>::TranslationType>(interpolatedTransEig);
148
149 Eigen::Quaternionf q1 = coloradar::internal::toEigenQuat(poses[tsIdx]);
150 Eigen::Quaternionf q2 = coloradar::internal::toEigenQuat(poses[tsIdx + 1]);
151 Eigen::Quaternionf interpolatedRotEig = q1.slerp(ratio, q2);
152 auto interpolatedRot = coloradar::internal::fromEigenQuat<typename PoseTraits<PoseT>::RotationType>(interpolatedRotEig);
153
154 PoseT interpolatedPose = coloradar::internal::makePose<PoseT>(interpolatedTrans, interpolatedRot);
155 interpolatedPoses.push_back(interpolatedPose);
156 }
157 return interpolatedPoses;
158}
