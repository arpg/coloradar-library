70Json::Value affineToJson(const Eigen::Affine3f& transform) {
71 Json::Value transformJson(Json::objectValue);
72 Eigen::Vector3f trans = transform.translation();
73 Json::Value transJson(Json::arrayValue);
74 transJson.append(trans.x());
75 transJson.append(trans.y());
76 transJson.append(trans.z());
77 transformJson["translation"] = transJson;
78 Eigen::Quaternionf rot(transform.linear());
79 Json::Value rotJson(Json::arrayValue);
80 rotJson.append(rot.x());
81 rotJson.append(rot.y());
82 rotJson.append(rot.z());
83 rotJson.append(rot.w());
84 transformJson["rotation"] = rotJson;
85 return transformJson;
86}

88std::filesystem::path ColoradarPlusDataset::exportToFile(DatasetExportConfig& exportConfig) {
89 Json::Value finalConfig;
90 finalConfig["runs"] = Json::arrayValue;
91 finalConfig["radar_config"] = cascadeConfig_->toJson();
92 finalConfig["data_content"] = Json::arrayValue;
93
94 auto exportRunNames = exportConfig.runs().empty() ? listRuns() : exportConfig.runs();
95 exportConfig.initDataset(exportRunNames);
96 std::vector<std::shared_ptr<ColoradarPlusRun>> exportRunObjects;
97 for (auto runName : exportRunNames) {
98 exportRunObjects.push_back(std::dynamic_pointer_cast<ColoradarPlusRun>(getRun(runName)));
99 finalConfig["runs"].append(runName);
100 }
101 std::cout << "Exporting " << finalConfig["runs"].size() << " run(s)." << std::endl;
102
103 H5::H5File datasetFile(exportConfig.destinationFilePath(), H5F_ACC_TRUNC);
104 std::vector<std::string> baseContent, imuContent, cascadeContent, lidarContent;
105 try {
106 baseContent = exportBaseDevice(exportConfig.base(), exportRunObjects, datasetFile);
107 } catch (const std::exception& e) {
108 std::cerr << "Error exporting base device data: " << e.what() << std::endl;
109 throw;
110 }
111 try {
112 imuContent = exportImu(exportConfig.imu(), exportRunObjects, datasetFile);
113 } catch (const std::exception& e) {
114 std::cerr << "Error exporting IMU data: " << e.what() << std::endl;
115 throw;
116 }
117 try {
118 cascadeContent = exportCascade(exportConfig.cascade(), exportRunObjects, datasetFile, &finalConfig);
119 } catch (const std::exception& e) {
120 std::cerr << "Error exporting cascade radar data: " << e.what() << std::endl;
121 throw;
122 }
123 try {
124 lidarContent = exportLidar(exportConfig.lidar(), exportRunObjects, datasetFile);
125 } catch (const std::exception& e) {
126 std::cerr << "Error exporting LiDAR data: " << e.what() << std::endl;
127 throw;
128 }
129
130 for (const auto &contentList : {baseContent, imuContent, cascadeContent, lidarContent}) {
131 if (contentList.empty()) continue;
132 for (const auto &content : contentList) {
133 finalConfig["data_content"].append(content);
134 }
135 }
136
137 if (exportConfig.exportTransforms()) {
138 std::cout << "Exporting transforms..." << std::endl;
139 Json::Value transformsJson(Json::objectValue);
140 transformsJson[transformBaseToCascadeContentName] = affineToJson(cascadeTransform_);
141 transformsJson[transformBaseToLidarContentName] = affineToJson(lidarTransform_);
142 transformsJson[transformBaseToImuContentName] = affineToJson(imuTransform_);
143 finalConfig["transforms"] = transformsJson;
144 }
145
146 std::string configString = Json::writeString(Json::StreamWriterBuilder(), finalConfig);
147 const hsize_t dims[1] = { static_cast<hsize_t>(configString.size()) };
148 H5::DataSpace dataspace(1, dims);
149 H5::DataSet configDataset = datasetFile.createDataSet("config", H5::PredType::STD_U8LE, dataspace);
150 const unsigned char* bytes = reinterpret_cast<const unsigned char*>(configString.data());
151 configDataset.write(bytes, H5::PredType::NATIVE_UINT8);
152 configDataset.close();
153
154 std::cout << "Closing file..." << std::endl;
155 datasetFile.close();
156 return exportConfig.destinationFilePath();
157}

162Eigen::Affine3f ColoradarPlusDataset::loadTransform(const std::filesystem::path& filePath) {
163 coloradar::internal::checkPathExists(filePath);
164
165 Eigen::Vector3f translation;
166 std::ifstream file(filePath);
167 std::string line;
168 std::getline(file, line);
169 std::istringstream iss(line);
170 iss >> translation.x() >> translation.y() >> translation.z();
171
172 Eigen::Quaternionf rotation;
173 std::getline(file, line);
174 iss.str(line);
175 iss.clear();
176 iss >> rotation.x() >> rotation.y() >> rotation.z() >> rotation.w();
177
178 Eigen::Affine3f transform = Eigen::Translation3f(translation) * rotation;
179 return transform;
180}

244std::vector<std::string> ColoradarPlusDataset::exportCascade(
245 const RadarExportConfig &config,
246 std::vector<std::shared_ptr<ColoradarPlusRun>> runs,
247 H5::H5File& datasetFile,
248 Json::Value* finalConfig
249) {
250 std::vector<std::string> content;
251 if (!config.exportTimestamps() && !config.exportPoses() && !config.exportDatacubes() && !config.exportHeatmaps() && !config.exportClouds()) return content;
252 std::cout << "Exporting cascade data..." << std::endl;
253
254 // Content types
255 if (config.exportTimestamps()) {
256 content.push_back(cascadeCubeTimestampsContentName);
257 content.push_back(cascadeTimestampsContentName);
258 }
259 if (config.exportPoses()) content.push_back(cascadePosesContentName);
260 if (config.exportClouds()) content.push_back(cascadeCloudsContentName);
261 if (config.exportDatacubes()) content.push_back(cascadeDatacubesContentName);
262 if (config.exportHeatmaps()) content.push_back(cascadeHeatmapsContentName);
263
264 // FOV
265 float range = cascadeConfig_->clipRange(config.heatmapCloudFov().rangeMeters);
266 int rangeMaxBin = cascadeConfig_->rangeToRangeIdx(range);
267 int azimuthMaxBin = config.heatmapCloudFov().azimuthIdx;
268 int elevationMaxBin = config.heatmapCloudFov().elevationIdx;
269 float horizontalFov = config.heatmapCloudFov().horizontalDegreesTotal;
270 float verticalFov = config.heatmapCloudFov().verticalDegreesTotal;
271 if (config.heatmapCloudFov().useDegreeConstraints) {
272 azimuthMaxBin = cascadeConfig_->horizontalFovToAzimuthIdx(config.heatmapCloudFov().horizontalDegreesTotal);
273 elevationMaxBin = cascadeConfig_->verticalFovToElevationIdx(config.heatmapCloudFov().verticalDegreesTotal);
274 }
275 else {
276 azimuthMaxBin = cascadeConfig_->clipAzimuthMaxBin(azimuthMaxBin);
277 elevationMaxBin = cascadeConfig_->clipElevationMaxBin(elevationMaxBin);
278 horizontalFov = cascadeConfig_->azimuthIdxToFovDegrees(azimuthMaxBin);
279 verticalFov = cascadeConfig_->elevationIdxToFovDegrees(elevationMaxBin);
280 }
281 std::shared_ptr<CascadeConfig> heatmapConfig;
282
283 for (auto run : runs) {
284 std::vector<double> timestamps = run->cascadeTimestamps();
285 hsize_t numFrames = timestamps.size();
286 std::vector<Eigen::Affine3f> basePoses = interpolatePoses(run->getPoses<Eigen::Affine3f>(), run->poseTimestamps(), timestamps);
287 std::vector<Eigen::Affine3f> sensorPoses(numFrames);
288
289 // timestamps
290 if (config.exportTimestamps()) {
291 coloradar::internal::saveVectorToHDF5(cascadeCubeTimestampsContentName + "_" + run->name(), datasetFile, run->cascadeCubeTimestamps());
292 coloradar::internal::saveVectorToHDF5(cascadeTimestampsContentName + "_" + run->name(), datasetFile, timestamps);
293 }
294
295 // poses
296 if (config.exportPoses()) {
297 coloradar::internal::savePosesToHDF5(cascadePosesContentName + "_" + run->name(), datasetFile, sensorPoses);
298 }
299
300 // datacubes
301 if (config.exportDatacubes()) {
302 hsize_t numCubeFrames = run->cascadeCubeTimestamps().size();
303 hsize_t datacubeSize = static_cast<hsize_t>(cascadeConfig_->numElevationBins() * cascadeConfig_->numAzimuthBins() * cascadeConfig_->nRangeBins() * 2);
304 std::vector<int16_t> datacubesFlat;
305 for (size_t i = 0; i < numCubeFrames; ++i) {
306 auto datacube = run->getCascadeDatacube(i);
307 datacubesFlat.insert(datacubesFlat.end(), datacube->begin(), datacube->end());
308 }
309 coloradar::internal::saveDatacubesToHDF5(cascadeDatacubesContentName + "_" + run->name(), datasetFile, datacubesFlat, numCubeFrames, datacubeSize);
310 }
311
312 // heatmaps
313 if (config.exportHeatmaps()) {
314 std::vector<float> heatmapsFlat;
315 for (size_t i = 0; i < numFrames; ++i) {
316 heatmapConfig = dynamic_pointer_cast<CascadeConfig>(cascadeConfig_->clone());
317 auto heatmap = run->getCascadeHeatmap(i);
318 auto heatmapResult = heatmapConfig->clipHeatmap(heatmap, azimuthMaxBin, elevationMaxBin, rangeMaxBin);
319 heatmap = heatmapResult.heatmap;
320 heatmapConfig = std::dynamic_pointer_cast<CascadeConfig>(heatmapResult.newConfig);
321 if (config.collapseElevation()) {
322 heatmapResult = heatmapConfig->collapseHeatmapElevation(heatmap, config.collapseElevationMinZ(), config.collapseElevationMaxZ());
323 heatmap = heatmapResult.heatmap;
324 heatmapConfig = std::dynamic_pointer_cast<CascadeConfig>(heatmapResult.newConfig);
325 }
326 if (config.removeDopplerDim()) {
327 heatmapResult = heatmapConfig->removeDoppler(heatmap);
328 heatmap = heatmapResult.heatmap;
329 heatmapConfig = std::dynamic_pointer_cast<CascadeConfig>(heatmapResult.newConfig);
330 }
331 heatmapsFlat.insert(heatmapsFlat.end(), heatmap->begin(), heatmap->end());
332 }
333 coloradar::internal::saveHeatmapsToHDF5(
334 cascadeHeatmapsContentName + "_" + run->name(),
335 datasetFile, heatmapsFlat, numFrames,
336 heatmapConfig->numAzimuthBins(),
337 heatmapConfig->nRangeBins(),
338 heatmapConfig->numElevationBins(),
339 heatmapConfig->hasDoppler()
340 );
341 heatmapsFlat.clear();
342 }
343 if (heatmapConfig) {
344 if (!finalConfig) throw std::runtime_error("exportCascade(): finalConfig is nullptr");
345 (*finalConfig)["heatmap_radar_config"] = heatmapConfig->toJson();
346 }
347
348 // clouds
349 if (config.exportClouds()) {
350 hsize_t numDims = cascadeConfig_->numElevationBins() > 0 ? 5 : 4; // x, y, (z), intensity, doppler
351 if (!cascadeConfig_->hasDoppler()) numDims -= 1;
352
353 bool buildClouds = false;
354 try {
355 auto cloud = run->getCascadePointcloud(0);
356 } catch (const std::filesystem::filesystem_error& e) {
357 buildClouds = true;
358 }
359 std::vector<float> cloudsFlat;
360 std::vector<hsize_t> cloudSizes(numFrames);
361 for (size_t i = 0; i < numFrames; ++i) {
362 std::shared_ptr<pcl::PointCloud<RadarPoint>> cloud;
363 if (buildClouds) {
364 cloud = cascadeConfig_->heatmapToPointcloud(run->getCascadeHeatmap(i), config.intensityThreshold());
365 }
366 else {
367 cloud = run->getCascadePointcloud(i, config.intensityThreshold());
368 }
369 filterFov(cloud, horizontalFov, verticalFov, range);
370 if (config.cloudsInGlobalFrame()) {
371 pcl::transformPointCloud(*cloud, *cloud, sensorPoses[i]);
372 }
373 if (config.collapseElevation()) {
374 collapseElevation(cloud, config.collapseElevationMinZ(), config.collapseElevationMaxZ());
375 }
376 cloudSizes[i] = cloud->size();
377 std::vector<float> cloudFlat = coloradar::internal::flattenRadarCloud(cloud, cascadeConfig_->numElevationBins(), cascadeConfig_->hasDoppler());
378 cloudsFlat.insert(cloudsFlat.end(), cloudFlat.begin(), cloudFlat.end());
379 }
380 coloradar::internal::saveCloudsToHDF5(cascadeCloudsContentName + "_" + run->name(), datasetFile, cloudsFlat, numFrames, cloudSizes, numDims);
381 }
382 }
383 datasetFile.flush(H5F_SCOPE_GLOBAL);
384 return content;
385}

387std::vector<std::string> ColoradarPlusDataset::exportLidar(const LidarExportConfig &config, std::vector<std::shared_ptr<ColoradarPlusRun>> runs, H5::H5File& datasetFile) {
388 std::vector<std::string> content;
389 if (!config.exportTimestamps() && !config.exportPoses() && !config.exportMap() && !config.exportMapSamples() && !config.exportClouds()) return content;
390 std::cout << "Exporting lidar data..." << std::endl;
391
392 // Content types
393 if (config.exportTimestamps()) content.push_back(lidarTimestampsContentName);
394 if (config.exportPoses()) content.push_back(lidarPosesContentName);
395 if (config.exportClouds()) content.push_back(lidarCloudsContentName);
396 if (config.exportMap()) content.push_back(lidarMapContentName);
397 if (config.exportMapSamples()) content.push_back(lidarMapSamplesContentName);
398
399 for (auto run : runs) {
400 std::vector<double> truePoseTimestamps = run->poseTimestamps();
401 std::vector<double> timestamps = run->lidarTimestamps();
402 hsize_t numFrames = timestamps.size();
403 auto truePoses = run->getPoses<Eigen::Affine3f>();
404 std::vector<Eigen::Affine3f> basePoses = interpolatePoses(truePoses, truePoseTimestamps, timestamps);
405 std::vector<Eigen::Affine3f> sensorPoses(numFrames);
406 for (int i = 0; i < numFrames; ++i) {
407 sensorPoses[i] = basePoses[i] * lidarTransform_;
408 }
409
410 // timestamps
411 if (config.exportTimestamps()) {
412 coloradar::internal::saveVectorToHDF5(lidarTimestampsContentName + "_" + run->name(), datasetFile, timestamps);
413 }
414
415 // poses
416 if (config.exportPoses()) {
417 coloradar::internal::savePosesToHDF5(lidarPosesContentName + "_" + run->name(), datasetFile, sensorPoses);
418 }
419
420 // clouds
421 if (config.exportClouds()) {
422 hsize_t numDims = config.collapseElevation() ? 3 : 4; // x, y, (z), intensity
423 if (config.removeIntensityDim()) numDims -= 1;
424 float range = config.cloudFov().rangeMeters > 0 ? config.cloudFov().rangeMeters : std::numeric_limits<float>::infinity();
425
426 std::vector<float> cloudsFlat;
427 std::vector<hsize_t> cloudSizes(numFrames);
428 for (size_t i = 0; i < numFrames; ++i) {
429 auto cloud = run->getLidarPointCloud(i);
430 filterFov(cloud, config.cloudFov().horizontalDegreesTotal, config.cloudFov().verticalDegreesTotal, range);
431 if (config.collapseElevation()) {
432 collapseElevation(cloud, config.collapseElevationMinZ(), config.collapseElevationMaxZ());
433 }
434 cloudSizes[i] = cloud->size();
435 auto cloudFlat = coloradar::internal::flattenLidarCloud(cloud, config.collapseElevation(), config.removeIntensityDim());
436 cloudsFlat.insert(cloudsFlat.end(), cloudFlat.begin(), cloudFlat.end());
437 }
438 coloradar::internal::saveCloudsToHDF5(lidarCloudsContentName + "_" + run->name(), datasetFile, cloudsFlat, numFrames, cloudSizes, numDims);
439 }
440
441 // map and samples
442 if (config.exportMap() || config.exportMapSamples()) {
443 pcl::PointCloud<pcl::PointXYZI>::Ptr map = std::make_shared<pcl::PointCloud<pcl::PointXYZI>>();
444 bool buildMap = false;
445 if (config.forceMapRebuild()) {
446 buildMap = true;
447 } else if (config.allowMapRebuild()) {
448 try {
449 map = run->getLidarOctomap();
450 } catch (...) {
451 std::cout << run->name() << ": lidar octomap not found." << std::endl;
452 buildMap = true;
453 }
454 }
455 if (buildMap) {
456 std::cout << run->name() << ": building octomap... " << std::endl;
457 octomap::OcTree octree = run->buildLidarOctomap(
458 config.mapResolution(),
459 config.mapInputCloudFov().horizontalDegreesTotal,
460 config.mapInputCloudFov().verticalDegreesTotal,
461 config.mapInputCloudFov().rangeMeters,
462 lidarTransform_
463 );
464 std::cout << run->name() << ": finished building octomap." << std::endl;
465 octreeToPcl(octree, map);
466 if (config.saveMap()) {
467 run->saveLidarOctomap(octree);
468 }
469 } else {
470 map = run->getLidarOctomap();
471 }
472
473 hsize_t numDims = config.collapseElevation() ? 3 : 4; // x, y, (z), occupancy
474 if (config.removeOccupancyDim()) numDims -= 1;
475 if (config.collapseElevation()) {
476 collapseElevation(map, config.collapseElevationMinZ(), config.collapseElevationMaxZ());
477 }
478 filterOccupancy(map, config.occupancyThresholdPercent() / 100.0, config.logOddsToProbability());
479
480 if (config.exportMap()) {
481 std::vector<float> mapFlat = coloradar::internal::flattenLidarCloud(map, config.collapseElevation(), config.removeOccupancyDim());
482 coloradar::internal::saveCloudToHDF5(lidarMapContentName + "_" + run->name(), datasetFile, mapFlat, numDims);
483 }
484
485 if (config.exportMapSamples()) {
486 std::vector<double> centerTimestamps;
487 Eigen::Affine3f centerTransform;
488 if (config.centerSensor()->name() == (new CascadeDevice())->name()) {
489 centerTimestamps = run->cascadeTimestamps();
490 centerTransform = cascadeTransform_;
491 } else if (config.centerSensor()->name() == (new LidarDevice())->name()) {
492 centerTimestamps = run->lidarTimestamps();
493 centerTransform = lidarTransform_;
494 } else if (config.centerSensor()->name() == (new ImuDevice())->name()) {
495 centerTimestamps = run->imuTimestamps();
496 centerTransform = imuTransform_;
497 } else {
498 centerTimestamps = run->poseTimestamps();
499 centerTransform = Eigen::Affine3f::Identity();
500 }
501 hsize_t numSamples = centerTimestamps.size();
502 std::vector<Eigen::Affine3f> basePosesCenterTs = interpolatePoses(truePoses, truePoseTimestamps, centerTimestamps);
503 std::vector<Eigen::Affine3f> centerPoses(numSamples);
504 for (int i = 0; i < numSamples; ++i) {
505 centerPoses[i] = basePosesCenterTs[i] * centerTransform;
506 }
507
508 bool resample = false;
509 if (config.forceResample()) {
510 resample = true;
511 } else if (config.allowResample()) {
512 try {
513 auto sample = run->getMapSample(0);
514 } catch (...) {
515 std::cout << run->name() << ": map samples not found, resampling... " << std::endl;
516 resample = true;
517 }
518 }
519
520 std::vector<float> samplesFlat;
521 std::vector<hsize_t> sampleSizes(numSamples);
522 for (size_t i = 0; i < numSamples; ++i) {
523 pcl::PointCloud<pcl::PointXYZI>::Ptr sample;
524 if (resample) {
525 sample = run->sampleMapFrame(
526 config.mapSampleFov().horizontalDegreesTotal,
527 config.mapSampleFov().verticalDegreesTotal,
528 config.mapSampleFov().rangeMeters,
529 centerPoses[i],
530 map
531 );
532 if (config.saveSamples()) {
533 run->saveMapSample(i, sample);
534 }
535 } else {
536 sample = run->getMapSample(i);
537 }
538 sampleSizes[i] = sample->size();
539 auto sampleFlat = coloradar::internal::flattenLidarCloud(sample, config.collapseElevation(), config.removeOccupancyDim());
540 samplesFlat.insert(samplesFlat.end(), sampleFlat.begin(), sampleFlat.end());
541 }
542 coloradar::internal::saveCloudsToHDF5(lidarMapSamplesContentName + "_" + run->name(), datasetFile, samplesFlat, numSamples, sampleSizes, numDims);
543 }
544 }
