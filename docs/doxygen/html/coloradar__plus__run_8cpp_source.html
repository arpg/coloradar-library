53std::vector<Eigen::Affine3f> ColoradarPlusRun::readPoses(const std::filesystem::path& path) const {
54 coloradar::internal::checkPathExists(path);
55 std::ifstream infile(path);
56 std::string line;
57 std::vector<Eigen::Affine3f> poses;
58
59 while (std::getline(infile, line)) {
60 float x, y, z, rotX, rotY, rotZ, rotW;
61 std::istringstream iss(line);
62 iss >> x >> y >> z >> rotX >> rotY >> rotZ >> rotW;
63 Eigen::Translation3f translation(x, y, z);
64 Eigen::Quaternionf rotation(rotW, rotX, rotY, rotZ);
65 Eigen::Affine3f pose = translation * rotation;
66 poses.push_back(pose);
67 }
68 return poses;
69}

99pcl::PointCloud<RadarPoint>::Ptr ColoradarPlusRun::readRadarPointcloud(
100 std::shared_ptr<RadarConfig> config,
101 const std::filesystem::path& binFilePath,
102 const double intensityThreshold
103) const {
104 std::ifstream file(binFilePath, std::ios::in | std::ios::binary);
105 if (!file.is_open()) {
106 throw std::filesystem::filesystem_error("Failed to open file: " + binFilePath.string(), std::make_error_code(std::errc::no_such_file_or_directory));
107 }
108 auto cloud = pcl::PointCloud<RadarPoint>::Ptr(new pcl::PointCloud<RadarPoint>());
109
110 float x, y, z, i, d;
111 while (true) {
112 if (!file.read(reinterpret_cast<char*>(&x), sizeof(float))) break;
113 if (!file.read(reinterpret_cast<char*>(&y), sizeof(float))) break;
114 if (!file.read(reinterpret_cast<char*>(&z), sizeof(float))) break;
115 if (!file.read(reinterpret_cast<char*>(&i), sizeof(float))) break;
116 if (!file.read(reinterpret_cast<char*>(&d), sizeof(float))) break;
117 if (i < intensityThreshold) continue;
118 cloud->points.emplace_back();
119 RadarPoint& pt = cloud->points.back();
120 pt.x = x; pt.y = y; pt.z = z;
121 pt.intensity = i;
122 pt.doppler = d;
123 }
124 cloud->width = static_cast<uint32_t>(cloud->size());
125 return cloud;
126}

128void ColoradarPlusRun::createRadarPointclouds(
129 const std::shared_ptr<RadarConfig>& config,
130 const std::filesystem::path& heatmapDirPath,
131 const std::filesystem::path& pointcloudDirPath,
132 double intensityThreshold)
133{
134 coloradar::internal::createDirectoryIfNotExists(pointcloudDirPath);
135 coloradar::internal::createDirectoryIfNotExists(pointcloudDirPath / "data");
136 const std::filesystem::path heatmapDataDir = heatmapDirPath / "data";
137
138 for (const auto& entry : std::filesystem::directory_iterator(heatmapDataDir)) {
139 const std::filesystem::path heatmapPath = entry.path();
140 if (!entry.is_regular_file() || heatmapPath.extension() != ".bin") continue;
141
142 auto heatmap = readHeatmap(heatmapPath, config);
143 auto cloud = config->heatmapToPointcloud(heatmap, intensityThreshold);
144
145 std::string filename = heatmapPath.filename().string();
146 const std::string heatmapStr = "heatmap";
147 filename.replace(filename.find(heatmapStr), heatmapStr.size(), "radar_pointcloud");
148 const std::filesystem::path cloudPath = pointcloudDirPath / "data" / filename;
149 std::ofstream file(cloudPath, std::ios::binary);
150 if (!file) throw std::runtime_error("Failed to open file: " + cloudPath.string());
151 file.exceptions(std::ofstream::failbit | std::ofstream::badbit);
152
153 for (const auto& pt : cloud->points) {
154 file.write(reinterpret_cast<const char*>(&pt.x), sizeof(float));
155 file.write(reinterpret_cast<const char*>(&pt.y), sizeof(float));
156 file.write(reinterpret_cast<const char*>(&pt.z), sizeof(float));
157 file.write(reinterpret_cast<const char*>(&pt.intensity), sizeof(float));
158 file.write(reinterpret_cast<const char*>(&pt.doppler), sizeof(float));
159 }
160 }
161}

227octomap::OcTree ColoradarPlusRun::buildLidarOctomap(
228 const double& mapResolution,
229 const float& lidarTotalHorizontalFov,
230 const float& lidarTotalVerticalFov,
231 const float& lidarMaxRange,
232 Eigen::Affine3f baseToLidarTransform
233) {
234 float maxRange = lidarMaxRange == 0 ? std::numeric_limits<float>::max() : lidarMaxRange;
235 std::vector<octomath::Pose6D> gtPoses = getPoses<octomath::Pose6D>();
236 std::vector<octomath::Pose6D> poses = interpolatePoses(gtPoses, poseTimestamps_, lidarTimestamps_);
237 auto baseToLidarT = coloradar::internal::fromEigenPose<octomath::Pose6D>(baseToLidarTransform);
238 octomap::OcTree tree(mapResolution);
239
240 for (size_t i = 0; i < lidarTimestamps_.size(); ++i) {
241 std::shared_ptr<OctoPointcloud> cloud = getLidarPointCloud<OctoPointcloud>(i);
242 cloud->filterFov(lidarTotalHorizontalFov, lidarTotalVerticalFov, maxRange);
243 auto frameTransform = poses[i] * baseToLidarT;
244 cloud->transform(frameTransform);
245 tree.insertPointCloud(*cloud, frameTransform.trans());
246 }
247 return tree;
248}

309void ColoradarPlusRun::createMapSamples(
310 const float horizontalFov,
311 const float verticalFov,
312 const float range,
313 const std::vector<double> sensorTimestamps,
314 const Eigen::Affine3f& baseToSensorTransform
315) {
316 for (const auto& entry : std::filesystem::directory_iterator(lidarMapsDirPath_)) {
317 if (entry.is_regular_file() && entry.path().filename() != "map.pcd") {
318 std::filesystem::remove(entry.path());

321 std::vector<Eigen::Affine3f> basePoses = getPoses<Eigen::Affine3f>();
322 if (!sensorTimestamps.empty()) basePoses = interpolatePoses(basePoses, poseTimestamps_, sensorTimestamps);
323 auto mapCloud = getLidarOctomap();
324
325 auto startT = std::chrono::high_resolution_clock::now();
326 for (size_t i = 0; i < basePoses.size(); ++i) {
327 Eigen::Affine3f mapToSensorT = basePoses[i] * baseToSensorTransform;
328 auto sample = Run::sampleMapFrame(horizontalFov, verticalFov, range, mapToSensorT, mapCloud);
329 saveMapSample(i, sample);

331 double elapsedTime = std::chrono::duration<double>(std::chrono::high_resolution_clock::now() - startT).count();
332 std::cout << std::fixed << std::setprecision(2) << name_ << ": sampled " << basePoses.size() << " map frames in " << elapsedTime << " seconds." << std::endl;
