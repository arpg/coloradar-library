4Eigen::Affine3f jsonToAffine(const Json::Value& transformJson) {
5 if (!transformJson.isObject() || !transformJson.isMember("translation") || !transformJson.isMember("rotation")) {
6 throw std::runtime_error("Invalid transform JSON: must be an object with 'translation' and 'rotation' keys.");
7 }
8 const Json::Value& transJson = transformJson["translation"];
9 if (!transJson.isArray() || transJson.size() != 3) {
10 throw std::runtime_error("Invalid translation JSON: must be an array of 3 floats.");
11 }
12 Eigen::Vector3f translation(transJson[0].asFloat(), transJson[1].asFloat(), transJson[2].asFloat());
13 const Json::Value& rotJson = transformJson["rotation"];
14 if (!rotJson.isArray() || rotJson.size() != 4) {
15 throw std::runtime_error("Invalid rotation JSON: must be an array of 4 floats (quaternion x,y,z,w).");
16 }
17 Eigen::Quaternionf rotation(rotJson[3].asFloat(), rotJson[0].asFloat(), rotJson[1].asFloat(), rotJson[2].asFloat());
18 Eigen::Affine3f transform = Eigen::Translation3f(translation) * rotation;
19 return transform;
20}

31 // read config
32 H5::DataSet datasetConfig = file.openDataSet("config");
33 H5::DataSpace space = datasetConfig.getSpace();
34 int ndims = space.getSimpleExtentNdims();
35 if (ndims != 1) throw std::runtime_error("HDF5 'config' dataset must be 1-D uint8.");
36 hsize_t dims[1] = {0};
37 space.getSimpleExtentDims(dims, nullptr);
38 std::vector<unsigned char> buf(static_cast<size_t>(dims[0]));
39 if (!buf.empty()) {
40 datasetConfig.read(buf.data(), H5::PredType::NATIVE_UINT8);

42 std::string configStr(reinterpret_cast<const char*>(buf.data()), buf.size());
43 Json::CharReaderBuilder b;
44 std::unique_ptr<Json::CharReader> reader(b.newCharReader());
45 Json::Value root;
46 std::string errs;
47 const char* begin = configStr.data();
48 const char* end = begin + configStr.size();
49 if (!reader->parse(begin, end, &root, &errs)) throw std::runtime_error("Failed to parse JSON from HDF5 'config': " + errs);
50 if (!root.isObject()) throw std::runtime_error("Invalid config JSON: expected an object at top level.");
51
52 // extract runs
53 std::vector<std::string> configRuns;
54 if (!root.isMember("runs") || !root["runs"].isArray()) {
55 throw std::runtime_error("Invalid config JSON: 'runs' must be an array.");
56 }
57 const Json::Value& runsNode = root["runs"];
58 configRuns.reserve(runsNode.size());
59 for (const auto& v : runsNode) {
60 if (!v.isString()) {
61 throw std::runtime_error("Invalid config JSON: 'runs' must be an array of strings.");
62 }
63 configRuns.push_back(v.asString());
64 }
65
66 // extract data_content
67 std::set<std::string, std::less<>> configDataContent;
68 if (!root.isMember("data_content") || !root["data_content"].isArray()) {
69 throw std::runtime_error("Invalid config JSON: 'data_content' must be an array.");
70 }
71 const Json::Value& dcNode = root["data_content"];
72 for (const auto& v : dcNode) {
73 if (!v.isString()) {
74 throw std::runtime_error("Invalid config JSON: 'data_content' must be an array of strings.");
75 }
76 configDataContent.insert(v.asString());
77 }
78
79 // extract cascade radar config
80 if (root.isMember("radar_config") && root["radar_config"].isObject()) {

82 }
83 if (root.isMember("heatmap_radar_config") && root["heatmap_radar_config"].isObject()) {
84 heatmapConfig_ = std::make_shared<coloradar::CascadeConfig>(root["heatmap_radar_config"]);
85 } else {

87 }
88

91 const Json::Value& transformsJson = root["transforms"];

93 cascadeTransform_ = jsonToAffine(transformsJson[transformBaseToCascadeContentName]);

98 if (transformsJson.isMember(transformBaseToImuContentName)) {

103 }
104
105 // read run data

112 std::vector<pcl::PointCloud<pcl::PointXYZI>::Ptr> lidarPointclouds = {};

114 std::vector<pcl::PointCloud<pcl::PointXYZI>::Ptr> mapSamples = {};

118 }
119 if (configDataContent.find(imuTimestampsContentName) != configDataContent.end()) {
