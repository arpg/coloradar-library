7void RadarConfig::precomputePointcloudTemplate() {
8 const size_t totalSize = numElevationBins_ * numAzimuthBins_ * nRangeBins();
9 pointcloudTemplate_.reset(new pcl::PointCloud<RadarPoint>);
10 pointcloudTemplate_->resize(totalSize);
11
12 std::vector<float> cosAzimuths(numAzimuthBins_), sinAzimuths(numAzimuthBins_);
13 for (int azIdx = 0; azIdx < numAzimuthBins_; ++azIdx) {
14 cosAzimuths[azIdx] = std::cos(azimuthBins_[azIdx]);
15 sinAzimuths[azIdx] = std::sin(azimuthBins_[azIdx]);
16 }
17 std::vector<float> cosElevations(numElevationBins_), sinElevations(numElevationBins_);
18 for (int elIdx = 0; elIdx < numElevationBins_; ++elIdx) {
19 cosElevations[elIdx] = std::cos(elevationBins_[elIdx]);
20 sinElevations[elIdx] = std::sin(elevationBins_[elIdx]);
21 }
22
23 size_t pointIdx = 0;
24 for (int elIdx = 0; elIdx < numElevationBins_; ++elIdx) {
25 for (int azIdx = 0; azIdx < numAzimuthBins_; ++azIdx) {
26 for (int rangeIdx = 0; rangeIdx < nRangeBins(); ++rangeIdx, ++pointIdx) {
27 float range = rangeIdx * rangeBinWidth_;
28 RadarPoint& point = (*pointcloudTemplate_)[pointIdx];
29 point.x = range * cosElevations[elIdx] * cosAzimuths[azIdx];
30 point.y = range * cosElevations[elIdx] * sinAzimuths[azIdx];
31 point.z = range * sinElevations[elIdx];
32 point.intensity = 0.0f;
33 point.doppler = 0.0f;
34 }
35 }
36 }

69HeatmapTransformResult RadarConfig::clipHeatmap(
70 const std::shared_ptr<std::vector<float>>& heatmap,
71 int azimuthMaxBin, int elevationMaxBin, int rangeMaxBin) const
72{
73 HeatmapTransformResult result;
74 result.heatmap = heatmap;
75 result.newConfig = this->clone();
76 azimuthMaxBin = clipAzimuthMaxBin(azimuthMaxBin);
77 elevationMaxBin = clipElevationMaxBin(elevationMaxBin);
78 rangeMaxBin = clipRangeMaxBin(rangeMaxBin);
79 if (azimuthMaxBin == numAzimuthBins_ / 2 && elevationMaxBin == numElevationBins_ / 2 && rangeMaxBin == nRangeBins() - 1) return result;
80
81 int azimuthBinLimit = numAzimuthBins_ / 2 - 1;
82 int azimuthLeftBin = azimuthBinLimit - azimuthMaxBin;
83 int azimuthRightBin = azimuthBinLimit + azimuthMaxBin + 1;
84
85 int elevationBinLimit = numElevationBins_ / 2 - 1;
86 int elevationLeftBin = elevationBinLimit - elevationMaxBin;
87 int elevationRightBin = elevationBinLimit + elevationMaxBin + 1;
88
89 result.heatmap = std::make_shared<std::vector<float>>();
90 for (int e = elevationLeftBin; e <= elevationRightBin; ++e) {
91 for (int a = azimuthLeftBin; a <= azimuthRightBin; ++a) {
92 for (int r = 0; r <= rangeMaxBin; ++r) {
93 int index = ((e * numAzimuthBins_ + a) * nRangeBins() + r) * 2;
94 if (hasDoppler_) {
95 for (int n = 0; n < 2; ++n) result.heatmap->push_back((*heatmap)[index + n]);
96 }
97 else result.heatmap->push_back((*heatmap)[index]);
98 }
99 }
100 }
101 result.newConfig->numAzimuthBins_ = azimuthRightBin - azimuthLeftBin + 1;
102 result.newConfig->numElevationBins_ = elevationRightBin - elevationLeftBin + 1;
103 result.newConfig->setNumRangeBins(rangeMaxBin + 1);
104 result.newConfig->azimuthBins_ = std::vector<float>(azimuthBins_.begin() + azimuthLeftBin, azimuthBins_.begin() + azimuthRightBin + 1);
105 result.newConfig->elevationBins_ = std::vector<float>(elevationBins_.begin() + elevationLeftBin, elevationBins_.begin() + elevationRightBin + 1);
106 return result;
107}

120HeatmapTransformResult RadarConfig::collapseHeatmapElevation(
121 const std::shared_ptr<std::vector<float>>& image,
122 const float& elevationMinMeters, const float& elevationMaxMeters) const
123{
124 if (elevationMaxMeters < elevationMinMeters) throw std::out_of_range("elevationMaxMeters must be greater or equal to elevationMinMeters.");
125 auto collapsedHeatmap = std::make_shared<std::vector<float>>(numAzimuthBins_ * nRangeBins() * 2);
126
127 for (int a = 0; a < numAzimuthBins_; ++a) {
128 for (int r = 0; r < nRangeBins(); ++r) {
129 float maxIntensity = -std::numeric_limits<float>::infinity();
130 float maxDoppler = 0.0f;
131 for (int e = 0; e < elevationBins_.size(); ++e) {
132 float z = r * std::sin(elevationBins_[e]);
133 if (z >= elevationMinMeters && z <= elevationMaxMeters) {
134 int index = (((e * numAzimuthBins_ + a) * nRangeBins() + r) * 2);
135 if ((*image)[index] > maxIntensity) {
136 maxIntensity = (*image)[index];
137 maxDoppler = (*image)[index + 1];
138 }
139 }
140 }
141 collapsedHeatmap->push_back(maxIntensity);
142 collapsedHeatmap->push_back(maxDoppler);
143 }
144 }
145 auto newConfig = this->clone();
146 newConfig->numElevationBins_ = 0;
147 newConfig->elevationBins_ = {};
148 HeatmapTransformResult result;
149 result.heatmap = collapsedHeatmap;
150 result.newConfig = newConfig;
151 return result;
152}

546std::map<std::string, std::vector<std::string>> readConfig(const std::filesystem::path& configFile) {
547 coloradar::internal::checkPathExists(configFile);
548 std::ifstream infile(configFile);
549 if (!infile.is_open()) {
550 throw std::runtime_error("Unable to open config file: " + configFile.string());
551 }
552 std::map<std::string, std::vector<std::string>> configMap;
553 std::string line;
554 std::regex keyValueRegex(R"(\s*([^#\s]+)\s*[:=\s]\s*(.*))");
555 const size_t maxKeySearchLength = 256;
556
557 while (std::getline(infile, line)) {
558 if (line.empty() || line.find_first_not_of(" \t") == std::string::npos || line.find('#') == 0) {
559 continue;
560 }
561 std::string lineHead = line.substr(0, maxKeySearchLength);
562 std::smatch match;
563 if (std::regex_match(lineHead, match, keyValueRegex)) {
564 std::string key = match[1];
565 size_t valueStartPos = match.position(2);
566 std::string valueSection = line.substr(valueStartPos);
567 std::istringstream valuesStream(valueSection);
568 std::string value;
569 std::vector<std::string> values;
570 while (valuesStream >> value) {
571 values.push_back(value);
572 }
573 configMap[key] = values;
574 }
575 }
576 infile.close();
577 return configMap;
578}

780void coloradar::RadarConfig::initPhaseFrequencyParams(const std::filesystem::path& phaseFrequencyCfgFile) {
781 const Json::Value& configMap = readJsonConfig(phaseFrequencyCfgFile);
782 if (!configMap.isMember("antennaCalib")) {
783 throw std::runtime_error("Missing antennaCalib in phase frequency config.");
784 }
785 const Json::Value& config = configMap["antennaCalib"];
786
787 std::vector<double> freqData(numTxAntennas_ * numRxAntennas_);
788 std::vector<std::complex<double>> phaseData(numTxAntennas_ * numRxAntennas_);
789
790 if (config.isMember("frequencySlope")) {
791 calibFrequencySlope_ = config["frequencySlope"].asDouble();
792 } else {
793 throw std::runtime_error("Missing frequencySlope in phase frequency config.");
794 }
795 if (config.isMember("samplingRate")) {
796 calibAdcSampleFrequency_ = config["samplingRate"].asInt();
797 } else {
798 throw std::runtime_error("Missing samplingRate in phase frequency config.");
799 }
800
801 if (config.isMember("frequencyCalibrationMatrix")) {
802 const Json::Value& frequencyMatrix = config["frequencyCalibrationMatrix"];
803 if (frequencyMatrix.size() != numTxAntennas_ * numRxAntennas_) {
804 throw std::runtime_error("Invalid frequency calibration array: expected " + std::to_string(numRxAntennas_ * numTxAntennas_) + " elements, got " + std::to_string(frequencyMatrix.size()));
805 }
806 int count = 0;
807 for (const auto& value : frequencyMatrix) {
808 freqData[count] = value.asDouble();
809 count++;
810 }
811 } else {
812 throw std::runtime_error("Missing frequencyCalibrationMatrix in phase frequency config.");
813 }
814
815 if (config.isMember("phaseCalibrationMatrix")) {
816 const Json::Value& phaseMatrix = config["phaseCalibrationMatrix"];
817 if (phaseMatrix.size() % 2 != 0) {
818 throw std::runtime_error("Invalid phaseCalibrationMatrix: Expecting pairs of real and imaginary values.");
819 }
820 if (phaseMatrix.size() / 2 != numTxAntennas_ * numRxAntennas_) {
821 throw std::runtime_error("Invalid phase calibration array: expected " + std::to_string(numRxAntennas_ * numTxAntennas_) + " elements, got " + std::to_string(phaseMatrix.size()));
822 }
823 for (Json::ArrayIndex i = 0; i < phaseMatrix.size(); i += 2) {
824 double real = phaseMatrix[i].asDouble();
825 double imag = phaseMatrix[i + 1].asDouble();
826 phaseData[i / 2] = std::complex<double>(real, imag);
827 }
828 } else {
829 throw std::runtime_error("Missing phaseCalibrationMatrix in phase frequency config.");
830 }
831
832 frequencyCalibMatrix_.clear();
833 frequencyCalibMatrix_.resize(numTxAntennas_ * numRxAntennas_ * numRangeBins_);
834 phaseCalibMatrix_.clear();
835 phaseCalibMatrix_.resize(numTxAntennas_ * numRxAntennas_);
836
837 for (int tx_idx = 0; tx_idx < numTxAntennas_; tx_idx++) {
838 for (int rx_idx = 0; rx_idx < numRxAntennas_; rx_idx++) {
839 int idx = rx_idx + (tx_idx * numRxAntennas_);
840 double delta_p = freqData[idx] - freqData[0];
841 double freq_calib = 2.0 * M_PI * delta_p / numRangeBins_ * (frequencySlope_ / calibFrequencySlope_) * (adcSampleFrequency_ / calibAdcSampleFrequency_);
842 for (int sample_idx = 0; sample_idx < numRangeBins_; sample_idx++) {
843 int cal_idx = sample_idx + numRangeBins_ * (rx_idx + numRxAntennas_ * tx_idx);
844 frequencyCalibMatrix_[cal_idx] = std::exp(std::complex<double>(0.0, -1.0) * std::complex<double>(freq_calib, 0.0) * std::complex<double>(sample_idx, 0.0));
845 }
846 }
847 }
848 std::complex<double> phase_ref = phaseData[0];
849 for (int tx_idx = 0; tx_idx < numTxAntennas_; tx_idx++) {
850 for (int rx_idx = 0; rx_idx < numRxAntennas_; rx_idx++) {
851 int idx = rx_idx + (tx_idx * numRxAntennas_);
852 phaseCalibMatrix_[idx] = phase_ref / phaseData[idx];
853 }
854 }

858void coloradar::RadarConfig::initInternalParams() {
859 azimuthApertureLen_ = 0;
860 elevationApertureLen_ = 0;
861 numVirtualElements_ = 0;
862 virtualArrayMap_.clear();
863 azimuthAngles_.clear();
864 elevationAngles_.clear();
865 azimuthAngles_.resize(numAzimuthBeams_);
866 elevationAngles_.resize(numElevationBeams_);
867 numAngles_ = numAzimuthBeams_ * numElevationBeams_;
868
869 for (int tx_idx = 0; tx_idx < numTxAntennas_; tx_idx++)
870 {
871 for (int rx_idx = 0; rx_idx < numRxAntennas_; rx_idx++)
872 {
873 int virtual_x = rxCenters_[rx_idx].x + txCenters_[tx_idx].x;
874 int virtual_y = rxCenters_[rx_idx].y + txCenters_[tx_idx].y;
875 // check to ensure this antenna pair doesn't map to the same virtual
876 // location as a previously evaluated antenna pair
877 bool redundant = false;
878 for (int i = 0; i < numVirtualElements_; i++)
879 {
880 int idx = i * 4;
881 if (virtualArrayMap_[idx] == virtual_x
882 && virtualArrayMap_[idx+1] == virtual_y)
883 redundant = true;
884 }
885 // record mapping from antenna pair index to virtual antenna location
886 // stored in vector with entries grouped into 4-tuples of
887 // [azimuth_location, elevation_location, rx_index, tx_index]
888 if (!redundant)
889 {
890 if (virtual_x + 1 > azimuthApertureLen_)
891 azimuthApertureLen_ = virtual_x + 1;
892 if (virtual_y + 1 > elevationApertureLen_)
893 elevationApertureLen_ = virtual_y + 1;
894
895 virtualArrayMap_.push_back(virtual_x);
896 virtualArrayMap_.push_back(virtual_y);
897 virtualArrayMap_.push_back(rx_idx);
898 virtualArrayMap_.push_back(tx_idx);
899 numVirtualElements_++;
900 }
901 }
902 }
903 double wavelength = c / (startFrequency_ + adcStartTime_ * frequencySlope_);
904 double chirp_time = idleTime_ + rampEndTime_;
905 double v_max = wavelength / (4.0 * numTxAntennas_ * chirp_time);
906 dopplerBinWidth_ = v_max / numDopplerBins_;
907
908 double center_frequency = startFrequency_ + numRangeBins_ / adcSampleFrequency_ * frequencySlope_ / 2.0;
909 double d = 0.5 * center_frequency / designFrequency_;
910 double az_d_phase = (2. * M_PI) / numAzimuthBeams_;
911 double phase_dif = (az_d_phase / 2.) - M_PI;
912 for (int i = 0; i < numAzimuthBeams_; i++) {
913 azimuthAngles_[i] = asin(phase_dif / (2. * M_PI * d));
914 phase_dif += az_d_phase;
915 }
916 double el_d_phase = (2.*M_PI) / numElevationBeams_;
917 phase_dif = (el_d_phase / 2.) - M_PI;
918 for (int i = 0; i < numElevationBeams_; i++) {
919 elevationAngles_[i] = asin(phase_dif / (2. * M_PI * d));
920 phase_dif += el_d_phase;
921 }
922}
