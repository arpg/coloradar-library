81template<coloradar::PointType PointT, coloradar::CloudType CloudT>
82std::shared_ptr<CloudT> coloradar::internal::readLidarPointCloud(const std::filesystem::path& binPath) {
83 coloradar::internal::checkPathExists(binPath);
84 std::ifstream infile(binPath, std::ios::binary);
85 if (!infile) {
86 throw std::runtime_error("Failed to open file: " + binPath.string());
87 }
88 infile.seekg(0, std::ios::end);
89 size_t numPoints = infile.tellg() / (4 * sizeof(float));
90 infile.seekg(0, std::ios::beg);
91
92 auto cloud = std::make_shared<CloudT>();
93 cloud->reserve(numPoints);
94
95 for (size_t j = 0; j < numPoints; ++j) {
96 float x, y, z, i;
97 infile.read(reinterpret_cast<char*>(&x), sizeof(float));
98 infile.read(reinterpret_cast<char*>(&y), sizeof(float));
99 infile.read(reinterpret_cast<char*>(&z), sizeof(float));
100 infile.read(reinterpret_cast<char*>(&i), sizeof(float));
101 PointT point = coloradar::internal::makePoint<PointT>(x, y, z, i);
102 cloud->push_back(point);
103 }
104 if (cloud->size() < 1) {
105 throw std::runtime_error("Failed to read or empty point cloud: " + binPath.string());
106 }
107 return cloud;
108}

111template<typename PointT, typename CloudT>
112void coloradar::internal::filterFov(std::shared_ptr<CloudT>& cloud, const float& horizontalFov, const float& verticalFov, const float& range) {
113 if (horizontalFov <= 0 || horizontalFov > 360) {
114 throw std::runtime_error("Invalid horizontal FOV value: expected 0 < FOV <= 360, got " + std::to_string(horizontalFov));
115 }
116 if (verticalFov <= 0 || verticalFov > 180) {
117 throw std::runtime_error("Invalid vertical FOV value: expected 0 < FOV <= 180, got " + std::to_string(verticalFov));
118 }
119 if (range <= 0) {
120 throw std::runtime_error("internal::filterFov(): Invalid max range value: expected R > 0, got " + std::to_string(range));
121 }
122 float horizontalHalfFovRad = horizontalFov / 2 * M_PI / 180.0f;
123 float verticalHalfFovRad = verticalFov / 2 * M_PI / 180.0f;
124 FovCheck<PointT> checkAzimuth = horizontalFov <= 180 ? checkAzimuthFrontOnly<PointT> : checkAzimuthFrontBack<PointT>;
125
126 CloudT unfilteredCloud(*cloud);
127 cloud->clear();
128
129 for (size_t i = 0; i < unfilteredCloud.size(); ++i) {
130 const PointT& point = unfilteredCloud[i];
131 float distance = std::sqrt(std::pow(getX(point), 2) + std::pow(getY(point), 2) + std::pow(getZ(point), 2));
132 if (distance > range) {
133 continue;
134 }
135 float elevationSin = std::sin(verticalHalfFovRad);
136 if (checkAzimuth(point, horizontalHalfFovRad) && (getZ(point) <= distance * elevationSin) && (getZ(point) >= -distance * elevationSin)) {
137 cloud->push_back(point);
138 }
139 }
140}
