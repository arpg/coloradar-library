88 H5::DataSet ds = file.openDataSet(name);
89 H5::DataSpace sp = ds.getSpace();
90 if (sp.getSimpleExtentNdims() != 1) {
91 throw std::runtime_error(name + ": expected rank-1 dataset");
92 }
93 hsize_t dim{};
94 sp.getSimpleExtentDims(&dim);
95 std::vector<T> v(static_cast<size_t>(dim));
